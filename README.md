    public boolean cuadroVacio(View view)
    {
        boolean vacio = false;

        if(campoTxtResultado.getText().equals(""))
            vacio = true;

        return vacio;
    }

    public void resultado(int val1, int val2)
    {

        switch (tipoOperacion)
        {
            case "suma":
                campoTxtResultado.setText("" + (val1+val2));
                break;
            case "resta":
                campoTxtResultado.setText("" + (val1-val2));
                break;
            case "multiplicacion":
                campoTxtResultado.setText("" + (val1*val2));
                break;
            case "division":
                campoTxtResultado.setText("" + (val1/val2));
        }
    }
